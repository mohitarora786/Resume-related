public function showTemplate($id)
    {
        $imagePath = public_path("Templates_Resumes/{$id}.png");
        if (!file_exists($imagePath)) {
            abort(404, 'Template not found.');
        }

        return view('resume-builder.template', [
            'id' => $id,
            'image' => asset("Templates_Resumes/{$id}.png")
        ]);
    }
